// Debounced write to Firestore so we don't spam it on every keystroke/drag
function saveState(patch = {}) {
  state = { ...state, ...patch, updatedAt: Date.now() };
  saveLocalCache(state);
  if (!currentUser) return;
  clearTimeout(saveTimer);
  saveTimer = setTimeout(async () => {
    try {
      justSavedAt = Date.now();
      await setDoc(userDocRef(currentUser.uid), state, { merge: true });
    } catch (err) {
      console.error(err);
      showToast('저장 중 오류가 발생했습니다.');
    }
  }, 500);
}

async function loadStateFromCloud(uid) {
  const snap = await getDoc(userDocRef(uid));
  if (snap.exists()) return snap.data();
  return {};
}

function listenToCloud(uid) {
  if (unsubscribeSnapshot) unsubscribeSnapshot();
  unsubscribeSnapshot = onSnapshot(userDocRef(uid), (snap) => {
    if (!snap.exists()) return;
    // Ignore the echo of our own recent write
    if (Date.now() - justSavedAt < 1500) return;
    const remote = snap.data();
    if (remote && remote.updatedAt && remote.updatedAt !== state.updatedAt) {
      state = remote;
      saveLocalCache(state);
      hydrateUI();
      showToast('다른 기기에서 업데이트된 내용을 불러왔습니다.');
    }
  });
}

function renderText(raw, title) {
  textTitle.textContent = title || '제목 없음';
  textSubtitle.textContent = 'Lector — 자유롭게 하이라이트하고 위치를 저장하세요';
  textContent.innerHTML = '';

  const lines = raw.split('\n');
  let para = [];
  lines.forEach(line => {
    const trimmed = line.trim();
    if (trimmed === '') {
      if (para.length) {
        flushPara(para.join(' '));
        para = [];
      }
    } else if (trimmed.startsWith('## ')) {
      if (para.length) { flushPara(para.join(' ')); para = []; }
      const ch = document.createElement('p');
      ch.className = 'chapter';
      ch.textContent = trimmed.slice(3);
      textContent.appendChild(ch);
    } else {
      para.push(trimmed);
    }
  });
  if (para.length) flushPara(para.join(' '));
}

function flushPara(text) {
  const p = document.createElement('p');
  p.textContent = text;
  textContent.appendChild(p);
}

function renderSavedMarker() {
  document.querySelectorAll('.pos-marker').forEach(el => el.remove());
  if (state.savedPos == null || !state.text) return;
  const total = document.documentElement.scrollHeight - window.innerHeight;
  if (!total) return;
  const ratio = state.savedPos / total;
  const contentH = textContent.scrollHeight;
  const targetY = contentH * ratio;

  const children = textContent.querySelectorAll('p');
  let bestEl = null, bestDist = Infinity;
  children.forEach(el => {
    const top = el.offsetTop;
    const dist = Math.abs(top - targetY);
    if (dist < bestDist) { bestDist = dist; bestEl = el; }
  });
  if (bestEl) {
    const marker = document.createElement('span');
    marker.className = 'pos-marker';
    bestEl.parentNode.insertBefore(marker, bestEl);
  }
}

function applyHighlight(sel, color) {
  // bugfix: guard against unknown colors (this is what made "red" swallow the
  // text — the span got class="hl red" but there was no CSS rule for it,
  // so it rendered with no background AND no forced text color safeguard)
  if (!VALID_COLORS.has(color)) color = 'yellow';

  const range = sel.getRangeAt(0);
  const text = sel.toString().trim();
  if (!text) return;

  const span = document.createElement('span');
  span.className = 'hl ' + color;
  const id = 'hl_' + Date.now();
  span.dataset.hlId = id;

  try {
    range.surroundContents(span);
  } catch {
    // multi-node selection fallback: extract and wrap (does not delete text,
    // just moves the selected nodes into the new span)
    const fragment = range.extractContents();
    span.appendChild(fragment);
    range.insertNode(span);
  }

  attachHlClick(span);

  const highlights = state.highlights ? [...state.highlights] : [];
  highlights.push({ id, color, text: text.slice(0, 300), time: Date.now() });
  saveState({ highlights });
  renderHlList();
  showToast('하이라이트를 추가했습니다.');
}

function attachHlClick(span) {
  span.addEventListener('dblclick', (e) => {
    e.stopPropagation();
    const id = span.dataset.hlId;
    const parent = span.parentNode;
    while (span.firstChild) parent.insertBefore(span.firstChild, span);
    parent.removeChild(span);
    parent.normalize();
    if (state.highlights) {
      saveState({ highlights: state.highlights.filter(h => h.id !== id) });
    }
    renderHlList();
    showToast('하이라이트를 삭제했습니다.');
  });
}

// ── HIGHLIGHT LIST ──
function renderHlList() {
  const hl = state.highlights || [];
  if (!hl.length) {
    hlList.innerHTML = '<div class="empty-hint">텍스트를 드래그하여<br>하이라이트를 추가하세요.<br><small style="color:#555">더블클릭으로 삭제</small></div>';
    return;
  }
  hlList.innerHTML = '';
  [...hl].reverse().forEach(h => {
    const card = document.createElement('div');
    card.className = 'bm-card';
    card.innerHTML = `
      <div class="bm-text">${escapeHtml(h.text)}</div>
      <div class="bm-meta">
        <span class="bm-dot ${h.color}"></span>
        <span>${new Date(h.time).toLocaleString('ko-KR', {month:'numeric',day:'numeric',hour:'2-digit',minute:'2-digit'})}</span>
      </div>
      <button class="bm-del" data-id="${h.id}" title="삭제">x</button>
    `;
    card.addEventListener('click', (e) => {
      if (e.target.classList.contains('bm-del')) return;
      const el = document.querySelector(`[data-hl-id="${h.id}"]`);
      if (el) {
        el.scrollIntoView({ behavior: 'smooth', block: 'center' });
        el.style.outline = '2px solid var(--accent)';
        setTimeout(() => el.style.outline = '', 1500);
      } else {
        showToast('해당 하이라이트를 찾을 수 없습니다.');
      }
    });
    card.querySelector('.bm-del').addEventListener('click', (e) => {
      e.stopPropagation();
      const id = e.target.dataset.id;
      const el = document.querySelector(`[data-hl-id="${id}"]`);
      if (el) {
        const parent = el.parentNode;
        while (el.firstChild) parent.insertBefore(el.firstChild, el);
        parent.removeChild(el);
        parent.normalize();
      }
      saveState({ highlights: (state.highlights || []).filter(h => h.id !== id) });
      renderHlList();
      showToast('삭제되었습니다.');
    });
    hlList.appendChild(card);
  });
}
